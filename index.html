            import 'dart:io';

void main() {
 // Tabla de pensiones por categoría
 Map pensiones = {
  'A': 550,
  'B': 500,
  'C': 460,
  'D': 400,
 };

 // Tabla de descuentos por nota promedio
 List> descuentos = [
  {'rango': [0.0, 13.99], 'descuento': 0},
  {'rango': [14.0, 15.99], 'descuento': 0.10},
  {'rango': [16.0, 17.99], 'descuento': 0.12},
  {'rango': [18.0, 20.0], 'descuento': 0.15},
 ];

 print('Ingresa la categoría del estudiante (A, B, C, D):');
 String? categoria = stdin.readLineSync()?.toUpperCase();

 // Verificar si la categoría es válida
 if (!pensiones.containsKey(categoria)) {
  print('Categoría inválida.');
  return;
 }

 // Solicitar nota promedio del estudiante
 print('Ingresa la nota promedio del estudiante:');
 double? promedio = double.tryParse(stdin.readLineSync()!);

 // Verificar si la nota promedio es válida
 if (promedio == null || promedio < 0 || promedio > 20) {
  print('Nota promedio inválida.');
  return;
 }

 // Obtener la pensión correspondiente a la categoría
 double pensionActual = pensiones[categoria]!;

 // Determinar el porcentaje de descuento según la nota promedio
 double descuentoAplicado = 0;
 for (var descuento in descuentos) {
  if (promedio >= descuento['rango'][0] && promedio <= descuento['rango'][1]) {
   descuentoAplicado = descuento['descuento'];
   break;
  }
 }

 // Calcular el monto de descuento y la nueva pensión
 double montoDescuento = pensionActual * descuentoAplicado;
 double nuevaPension = pensionActual - montoDescuento;

 // Mostrar resultados
 print('La pensión actual es: S/ $pensionActual');
 print('El descuento aplicado es: ${descuentoAplicado * 100}%');
 print('El monto del descuento es: S/ $montoDescuento');
 print('La nueva pensión es: S/ $nuevaPension');
}

import 'dart:io';

void main() {
  // Precios base para mayores y jóvenes
  double precioMayor = 350;
  double precioJoven = 200;

  // Descuentos por nivel para mayores de 18 años
  Map descuentoMayor = {
    1: 0.40,
    2: 0.30,
    3: 0.15,
    4: 0.0, // Sin descuento para otros niveles
  };

  // Descuentos por nivel para jóvenes de 18 años
  Map descuentoJoven = {
    1: 0.60,
    2: 0.40,
    3: 0.20,
    4: 0.0, // Sin descuento para otros niveles
  };

  // Solicitar la edad del hombre
  print('Ingresa la edad del hombre:');
  int? edad = int.tryParse(stdin.readLineSync()!);

  if (edad == null || edad < 0) {
    print('Edad inválida.');
    return;
  }

  // Solicitar el nivel de sistema de beneficio
  print('Ingresa el nivel de sistema de beneficio (1, 2, 3, 4):');
  int? nivel = int.tryParse(stdin.readLineSync()!);

  if (nivel == null || !descuentoMayor.containsKey(nivel)) {
    print('Nivel inválido.');
    return;
  }

  double precioFinal = 0;
  double descuentoAplicado = 0;

  // Calcular según la edad
  if (edad > 18) {
    // Hombres mayores de 18 años
    descuentoAplicado = descuentoMayor[nivel]!;
    precioFinal = precioMayor * (1 - descuentoAplicado);
  } else if (edad == 18) {
    // Hombres de 18 años
    descuentoAplicado = descuentoJoven[nivel]!;
    precioFinal = precioJoven * (1 - descuentoAplicado);
  } else {
    print('No se aplica la venta de libreta para menores de 18 años.');
    return;
  }

  // Mostrar resultados
  print('El descuento aplicado es: ${descuentoAplicado * 100}%');
  print('El valor final a pagar es: S/ $precioFinal');
}

Código N° 01 - EJERCICIO SUMATORIA DE NÙMEROS PRIMOS EN UN RANGO EN EL LENGUAJE DE DART


            import 'dart:io';

            bool esPrimo(int numero) {
             if (numero < 2) {
              return false;
             }
             for (int i = 2; i <= numero ~/ 2; i++) {
              if (numero % i == 0) {
               return false;
              }
             }
             return true;
            }
            
            int sumarPrimosEnRango(int inicio, int fin) {
             int suma = 0;
             for (int i = inicio; i <= fin; i++) {
              if (esPrimo(i)) {
               suma += i;
              }
             }
             return suma;
            }
            
            void main() {
             stdout.write('Ingresa el número inicial: ');
             int inicio = int.parse(stdin.readLineSync()!);
            
             stdout.write('Ingresa el número final: ');
             int fin = int.parse(stdin.readLineSync()!);
            
             int sumaPrimos = sumarPrimosEnRango(inicio, fin);
             print('La suma de los números primos entre $inicio y $fin es: $sumaPrimos');
            }

            import 'dart:math';
            import 'dart:core';
            import 'dart:typed_data';
            
            // Función para calcular el factorial utilizando BigInt para manejar números grandes
            BigInt calcularFactorial(int n) {
              BigInt resultado = BigInt.one;
              
              for (int i = 1; i <= n; i++) {
                resultado *= BigInt.from(i);
              }
              
              return resultado;
            }
            
            void main() {
              int numero = 100;  // Puedes cambiar el número si lo deseas
              
              BigInt factorial = calcularFactorial(numero);
              
              print("El factorial de $numero es: $factorial");

Código N° 04 - EJERCICIO SOBRE LA INVERSIÓN DE UN NÚMERO EN EL LENGUAJE DE DART


            import 'dart:io';

void main() {
  // Solicitar al usuario que ingrese un número entero
  stdout.write("Ingrese un número entero: ");
  int numero = int.parse(stdin.readLineSync()!);

  // Inicializar la variable para almacenar el número invertido
  int invertido = 0;

  // Usar el bucle while para invertir el número
  while (numero != 0) {
    int digito = numero % 10;  // Extraer el último dígito
    invertido = invertido * 10 + digito;  // Agregar el dígito al número invertido
    numero = numero ~/ 10;  // Eliminar el último dígito del número original
  }

  // Mostrar el resultado
  print("El número invertido es: $invertido");
}

Código N° 05 - EJERCICIO SOBRE LA SUMA DE MATRICES NXN EN EL LENGUAJE DE DART


            import 'dart:io';

            void main() {
              // Solicitar el tamaño de las matrices N x N
              stdout.write("Ingrese el tamaño de las matrices (N): ");
              int n = int.parse(stdin.readLineSync()!);
            
              // Crear dos matrices vacías de tamaño N x N
              List> matriz1 = List.generate(n, (_) => List.filled(n, 0));
              List> matriz2 = List.generate(n, (_) => List.filled(n, 0));
              List> sumaMatriz = List.generate(n, (_) => List.filled(n, 0));
            
              // Solicitar valores para la primera matriz
              print("Ingrese los valores de la primera matriz:");
              for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                  stdout.write("Elemento [$i][$j]: ");
                  matriz1[i][j] = int.parse(stdin.readLineSync()!);
                }
              }
            
              // Solicitar valores para la segunda matriz
              print("Ingrese los valores de la segunda matriz:");
              for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                  stdout.write("Elemento [$i][$j]: ");
                  matriz2[i][j] = int.parse(stdin.readLineSync()!);
                }
              }
            
              // Sumar las dos matrices
              for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                  sumaMatriz[i][j] = matriz1[i][j] + matriz2[i][j];
                }
              }
            
              // Mostrar el resultado de la suma de las matrices
              print("La suma de las matrices es:");
              for (int i = 0; i < n; i++) {
                for (int j = 0; j < n; j++) {
                  stdout.write("${sumaMatriz[i][j]} ");
                }
                print("");  // Nueva línea después de cada fila
              }
            }

Código N° 07 - EJERCICIO SOBRE MATRÍZ DE ESPIRAL EN EL LENGUAJE DE DART


        void main() {
            int n = 5; // Cambia este valor para probar con diferentes tamaños de matriz
          
            List> matriz = List.generate(n, (_) => List.filled(n, 0));
          
            int valor = 1;
            int filaInicio = 0, filaFin = n - 1;
            int colInicio = 0, colFin = n - 1;
          
            // Bucle para llenar la matriz en espiral
            while (filaInicio <= filaFin && colInicio <= colFin) {
              // Llenar la fila superior (de izquierda a derecha)
              for (int i = colInicio; i <= colFin; i++) {
                matriz[filaInicio][i] = valor++;
              }
              filaInicio++;
          
              // Llenar la columna derecha (de arriba hacia abajo)
              for (int i = filaInicio; i <= filaFin; i++) {
                matriz[i][colFin] = valor++;
              }
              colFin--;
          
              // Llenar la fila inferior (de derecha a izquierda)
              if (filaInicio <= filaFin) {
                for (int i = colFin; i >= colInicio; i--) {
                  matriz[filaFin][i] = valor++;
                }
                filaFin--;
              }
          
              // Llenar la columna izquierda (de abajo hacia arriba)
              if (colInicio <= colFin) {
                for (int i = filaFin; i >= filaInicio; i--) {
                  matriz[i][colInicio] = valor++;
                }
                colInicio++;
              }
            }
          
            // Imprimir la matriz en espiral
            print("Matriz en espiral de tamaño $n:");
            for (var fila in matriz) {
              print(fila);
            }
          }

Código N° 08 - EJERCICIO SOBRE VERIFICACIÓN DE UN NÚMERO ARMSTRONG EN EL LENGUAJE DE DART


        import 'dart:math'; // Para usar pow()
        import 'dart:io';   // Para recibir entrada del usuario
        
        void main() {
          // Solicitar al usuario que ingrese un número
          print("Ingresa un número para verificar si es un número Armstrong:");
          int numero = int.parse(stdin.readLineSync()!);
        
          // Convertir el número en una lista de dígitos
          String numeroStr = numero.toString();
          int n = numeroStr.length; // Número de dígitos
        
          // Inicializar la suma
          int suma = 0;
        
          // Bucle para separar y elevar cada dígito
          for (int i = 0; i < n; i++) {
            int digito = int.parse(numeroStr[i]);
            suma += pow(digito, n).toInt(); // Elevar a la potencia n
          }
        
          // Verificar si el número es Armstrong
          if (suma == numero) {
            print("$numero es un número Armstrong.");
          } else {
            print("$numero no es un número Armstrong.");
          }
        }

Código N° 09 - EJERCICIO SOBRE ÉL CÁLCULO DE POTENCIAS USANDO MULTIPLICACIÓN REPETIDA EN EL LENGUAJE DE DART


        import 'dart:io'; // Para recibir entrada del usuario

        void main() {
          // Solicitar la base y el exponente al usuario
          print("Ingresa la base:");
          int base = int.parse(stdin.readLineSync()!);
          
          print("Ingresa el exponente:");
          int exponente = int.parse(stdin.readLineSync()!);
          
          // Inicializar el resultado en 1 (cualquier número elevado a 0 es 1)
          int resultado = 1;
          
          // Usar un bucle for para realizar la multiplicación repetida
          for (int i = 0; i < exponente; i++) {
            resultado *= base;
          }
          
          // Mostrar el resultado
          print("El resultado de $base^$exponente es: $resultado");
        }
